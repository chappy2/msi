addHttpHeader=(req, res, next) ->
    matchUrl = '/clientApp.js';
    if (req.url.substring(0, matchUrl.length) == matchUrl) 
        res.setHeader "X-SourceMap", "/clientApp.js.map"
    return next()  

app.use addHttpHeader
